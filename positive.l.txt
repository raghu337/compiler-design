%{
#include <stdio.h>
int pos_count = 0;
int neg_count = 0;
%}

%%
-([0-9]+)      { neg_count++; }      /* Negative numbers */
([0-9]+)       { pos_count++; }      /* Positive numbers */
[ \t]+         ;                     /* Ignore spaces */
\n             ;                     /* Ignore newline */
.              ;                     /* Ignore other characters */
%%

int yywrap(void)
{
    return 1;
}

int main(void)
{
    printf("Enter numbers:\n");
    yylex();

    printf("\nPositive numbers: %d\n", pos_count);
    printf("Negative numbers: %d\n", neg_count);

    return 0;
}
